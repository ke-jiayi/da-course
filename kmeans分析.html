在线运行代码

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

# 模拟数据
np.random.seed(42)
n_users = 1000
user_feat = pd.DataFrame({
    'pv': np.random.randint(1,50,n_users),
    'cart': np.random.randint(0,10,n_users),
    'fav': np.random.randint(0,8,n_users),
    'buy': np.random.randint(0,5,n_users)
})

# 聚类
scaler = StandardScaler()
X = scaler.fit_transform(user_feat)
kmeans = KMeans(4, random_state=42)
user_feat['label'] = kmeans.fit_predict(X)

# 字体（兼容网页）
plt.rcParams['font.sans-serif'] = ['DejaVu Sans']

# 散点图
plt.figure(figsize=(8,6))
plt.scatter(user_feat.pv, user_feat.buy, c=user_feat.label, cmap='viridis', alpha=0.6)
plt.savefig('/tmp/scatter.png')
plt.close()

# 柱状图
plt.figure(figsize=(8,6))
user_feat.groupby('label')[['pv','buy']].mean().plot(kind='bar')
plt.savefig('/tmp/bar.png')
plt.close()

# 雷达图（简化）
plt.figure(figsize=(8,6))
plt.bar([1,2,3,4], [3,2,4,1])
plt.savefig('/tmp/radar.png')
plt.close()

print("✅ 图表生成完成")
